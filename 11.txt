CREATE OR REPLACE TRIGGER STOP
BEFORE UPDATE OR DELETE OR INSERT ON EMPLOYEE
FOR EACH ROW
    DECLARE
        E EXCEPTION;
        TODAY DATE:=SYSDATE;
        CURRENT VARCHAR2(20);
    BEGIN
        CURRENT:=TO_CHAR(TODAY,'DAY');
        CURRENT:=RTRIM(CURRENT);
        IF CURRENT='THURSDAY' OR CURRENT='FRIDAY' THEN
            RAISE E;
        END IF;
    EXCEPTION
        WHEN E THEN
        DBMS_OUTPUT.PUT_LINE('You shall not pass');
    END;
    /
